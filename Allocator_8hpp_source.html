156 void SetupFunctionPointers()
157 {
158 allocateFn = [](void *obj, size_t size, size_t alignment, const std::source_location &location) -> void *
159 {
160 if constexpr (HasAllocate<T>)
161 {
162 return reinterpret_cast<T *>(obj)->Allocate(size, alignment, location);
163 }
164 return nullptr;
165 };

167 deallocateFn = [](void *obj, void *ptr)

173 };
174
175 deallocateAllFn = [](void *obj)
